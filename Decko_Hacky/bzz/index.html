var state;

function hack(){
    state.stems.forEach((a, b)=>state.stems[b]=true)
    state.treasures.forEach((a, b)=>state.treasures[b]=true)
    state.decorations.forEach((a, b)=>state.decorations[b]=true)
    state.plants.forEach((a, b)=>state.plants[b]=true)
    //state.moneyEarned = 1e+100
    dpapi.saveAppState(1, "levelData_1", [{name: "levelData_1", state: JSON.stringify(state)}], integrativeContainer.getToken())
}

dpapi.loadAppState(1, "levelData_1", integrativeContainer.getToken()).then(a=>{state=a;hack()})
